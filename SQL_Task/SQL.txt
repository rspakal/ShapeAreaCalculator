Products: содержит столбцы ProductID и ProductName
Categories: содержит столбцы CategoryID и CategoryName
ProductCategories: содержит столбцы ProductID и CategoryID

SELECT 
    p.ProductName, 
    c.CategoryName
FROM 
    Products p
LEFT JOIN 
    ProductCategories pc ON p.ProductID = pc.ProductID
LEFT JOIN 
    Categories c ON pc.CategoryID = c.CategoryID
